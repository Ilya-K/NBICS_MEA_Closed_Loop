44  public TData[] SE(TRawData[] data)
45  {
46  int length = data.Length;
47  int nBlocks = (length + partialN) / N_SAMPLES;
48  int reminder = (length + partialN) % N_SAMPLES;
49  TData[] result = new TData[nBlocks];
50 
51  int counter = 0;
52  int shift = 0;
53  while (true)
54  {
55  if (counter == nBlocks) // Process final block if exists (reminder > 0)
56  {
57  partialN = reminder;
58  for (int i = shift; i < shift + partialN; ++i)
59  {
60  partialSum += data[i];
61  partialSum2 += ((uint)data[i] * (uint)data[i]);
62  }
63  break; // and exit
64  }
65 
66  int nSamples = N_SAMPLES;
67  if (counter == 0) nSamples -= partialN; // Process incomplete buffer first
68 
69  for (int i = shift; i < shift + nSamples; ++i) // Process full length buffers in the middle
70  {
71  partialSum += data[i];
72  partialSum2 += ((uint)data[i] * (uint)data[i]);
73  }
74  shift += nSamples;
75 
76  // SE = sqrt(mean(X^2) - (mean(X))^2)
77  result[counter++] = Math.Sqrt(((TData)partialSum2 - (TData)((UInt64)partialSum * partialSum) / N_SAMPLES) / N_SAMPLES);
78  partialSum = 0;
79  partialSum2 = 0;
80  }
81  return result;
