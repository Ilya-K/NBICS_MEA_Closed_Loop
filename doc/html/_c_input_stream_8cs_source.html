218  public UInt32 GetIntervalFromNowInMS(TTime futureTime)
219  {
220  UInt32 delta;

249  // Runs in separate stream
250  // Reads out queue ant calls hahdlers for each data packet
251  private void RunStream()
252  {
253  TRawDataPacket dataPacket = null;
254  do
255  {
256  m_notEmpty.WaitOne();
257  if (!m_stop)
258  {
259  lock (m_rawQueue)
260  {
261  if (m_rawQueue.Count > 0)
262  {
263  dataPacket = m_rawQueue.Dequeue();
264 
265  lock (lockDeqTime) m_deqTimeStamp += (ulong)m_lastBufLength;
266  m_lastBufLength = dataPacket.First().Value.Length;
267 
268  if (m_rawQueue.Count > 0) m_notEmpty.Set();
269  }
270  }
271  if (dataPacket != null)
272  lock (ConsumerList)
273  if ((ConsumerList != null) && (ConsumerList.Count != 0))
274  foreach (ConsumerDelegate consumer in ConsumerList) consumer(dataPacket);
275  }
276  else
277  {
278  if (!m_stopped)
279  {
280  m_dataSource.StopDacq();
281  m_stopped = true;
282  }
283  }
284  } while (!m_kill);
285  }
