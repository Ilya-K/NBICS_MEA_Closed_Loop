415  private void update_X0123()
416  {
417  double y_new = source[t0 + tau];
418  double y_old = source[t0 - tau - 1];
419  X0 += y_new - y_old;
420  X1 += (double)tau_plus_1 * y_new - (double)minus_tau * y_old - X0;
421  X2 += (double)tau_plus_1_squared * y_new - (double)minus_tau_squared * y_old - X0 - 2 * X1;
422  X3 += (double)tau_plus_1_cubed * y_new - (double)minus_tau_cubed * y_old - X0 - 3 * X1 - 3 * X2;
423  }

429  X0 += (y_new - y_old);
