263  private bool ParseHeader(string fileName, ref int sampleRate, ref int zeroLevel, ref long startOfData, ref int[] elArray)
264  {
265  using (StreamReader strReader = new StreamReader(fileName))
266  {
267  char[] header = new char[11];
268  strReader.Read(header, 0, 11);
269  int position = 11;
270 
271  if (new string(header).CompareTo("MC_DataTool") == 0)
272  {
273  string s;
274  s = strReader.ReadLine(); // Rest of the first line
275  position += s.Length + CRLF;
276  s = strReader.ReadLine(); // "Version" line
277  position += s.Length + CRLF;
278  s = strReader.ReadLine(); // "MC_REC file" line
279  position += s.Length + CRLF;
280 
281  // Sample Rate
282  s = strReader.ReadLine();
283  position += s.Length + CRLF;
284  int start = s.LastIndexOf('=') + 1;
285  sampleRate = Int32.Parse(s.Substring(start));
286 
287  // ADC Zero Level
288  s = strReader.ReadLine();
289  position += s.Length + CRLF;
290  start = s.LastIndexOf('=') + 1;
291  zeroLevel = Int32.Parse(s.Substring(start));
292 
293  // Voltage (uV/bit)
294  s = strReader.ReadLine();
295  position += s.Length + CRLF;
296 
297  // List of Channels
298  s = strReader.ReadLine().Substring(START_CHANNEL_LIST);
299  position += s.Length + START_CHANNEL_LIST + CRLF;
300  string[] elNumbers = s.Split(new string[] { ";El_" }, MEA.MAX_CHANNELS, StringSplitOptions.RemoveEmptyEntries);
301  elArray = Array.ConvertAll(elNumbers, el => MEA.EL_DECODE[Int32.Parse(el)]);
302 
303  // End Of Header
304  if (strReader.ReadLine() == "EOH")
305  {
306  startOfData = position + 5;
307  }
308  else
309  {
310  throw new Exception("Wrong file format!");
311  }
312  return true;
313  }
314  else
315  {
316  return false;
317  }
318  }
319  }
